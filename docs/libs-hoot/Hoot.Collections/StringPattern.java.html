  public    StringPattern(final CharacterString pattern, final CharacterString wildCard)
  {
    java.lang.String exitID = "StringPattern>>StringPattern";
    Frame f0 = new Frame(exitID);
    java.lang.String p = pattern.primitiveString();

    java.lang.String w = wildCard.primitiveString();

    StringTokenizer tokens = Primitive.tokens(p, w);

    int count = tokens.countTokens();
    if ((0 < count)) {
      if (p.startsWith(w)) {
        count += 1;
      };
      if (p.endsWith(w)) {
        count += 1;
      };
      if (p.startsWith(w)) {
        segments.add(EmptyString.primitiveString());
        count += 1;
      };
      while(tokens.hasMoreTokens()) {
        segments.add(tokens.nextToken());
        count += 1;
      };
      if (p.endsWith(w)) {
        segments.add(EmptyString.primitiveString());
        count += 1;
      };
    };
  }

  /**

  public   Boolean accepts(final CharacterString aString)
  {
    java.lang.String exitID = "StringPattern>>accepts";
    Frame f0 = new Frame(exitID);
    java.lang.String sample = aString.primitiveString();

    int count = segments.size();
    if ((0 > count)) {
      return True.literal();
    };
    if (!(sample.startsWith(segments.get(0)))) {
      return False.literal();
    };
    if ((0 == count)) {
      return Boolean.from((sample.length() == segments.get(0).length()));
    };
    if (!(sample.endsWith(segments.get(count)))) {
      return False.literal();
    };
    if ((1 < count)) {
      int x = 0;

      int p = segments.get(0).length();
      while((count > x)) {
        java.lang.String s = segments.get(x);
        if ((0 < x)) {
          int a = sample.indexOf(s, p);
          if ((0 > a)) {
            return False.literal();
          }
          else {
             p = (a + s.length());
          };
        };
        x += 1;
      };
    };
    return (Boolean)True.literal();
  }
